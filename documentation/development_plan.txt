```
    Set random seed for Python random, numpy, torch (CPU + CUDA).
```

```
    """
```

### TASK T07: Preprocessing

**Dependencies**: T01, T03, T05
**Spec reference**: Section 5 (Target Variable Handling), Section 6
(Minimal Preprocessing)

**Objective**: Implement `hybrid_embed/data/preprocess.py` with
all preprocessing logic: numeric scaling, missing value imputation,
categorical encoding, and target variable handling.

**Deliverables**:
- `hybrid_embed/data/preprocess.py`
- `tests/data/test_preprocess.py`

**Detailed requirements**:

The preprocessor must be fitted on training data only, then applied
(transformed) to val/test data. This is the sklearn fit/transform
pattern.

```
class TabularPreprocessor:
    """Preprocess tabular data for embedding model consumption.

    Fitted on training data only. Transforms train/val/test
    consistently.

    Parameters
    ----------
    schema : Schema
        Inferred column types.
    scaler_type : str, default "standard"
        "standard" or "robust".
    max_categorical_cardinality : int, default 1000
        Cap vocabulary for high-cardinality categoricals.
    add_missing_indicator : bool, default False
        If True, add binary indicator columns for missingness.
    """

    def fit(self, X_train: pd.DataFrame, y_train: np.ndarray = None) -> "TabularPreprocessor":
        """Fit preprocessor on training data.

        Learns:
        - Median per numeric column (for imputation)
        - Scaler parameters (StandardScaler or RobustScaler)
        - Vocabulary per categorical column (with cardinality cap)
        - Target encoder/scaler (if y_train provided)

        Returns self.
        """

    def transform(self, X: pd.DataFrame) -> dict:
        """Transform a DataFrame into model-ready format.

        Returns
        -------
        dict with keys:
            "numeric": np.ndarray of shape (n_samples, n_numeric)
                Scaled, imputed numeric features.
            "categorical": np.ndarray of shape (n_samples, n_categorical)
                Integer-encoded categorical features (indices into
                vocabularies). dtype=int64.
        """

    def transform_target(self, y: np.ndarray) -> np.ndarray:
        """Transform target variable.

        - Classification: LabelEncoder (string/non-contiguous -> 0..K-1)
        - Regression: StandardScaler
        """

    def inverse_transform_target(self, y: np.ndarray) -> np.ndarray:
        """Inverse-transform target (for regression predictions)."""

    def get_n_categories_per_column(self) -> list[int]:
        """Return vocabulary size per categorical column.

        Includes +1 for __OOV__ and +1 for __MISSING__ tokens.
        Used by embedding layers to size their embedding tables.
        """

    def get_n_numeric_features(self) -> int:
        """Return number of numeric features after preprocessing."""

    @property
    def n_classes(self) -> int | None:
        """Number of classes (classification) or None (regression)."""
```

**Key implementation details**:

- Numeric imputation: fill NaN with train-fold median per column.
- Numeric scaling: fit StandardScaler (or RobustScaler) on imputed
  train data.
- Categorical encoding: build vocabulary from train fold. Map each
  value to an integer index. Reserve index 0 for `__OOV__`, index 1
  for `__MISSING__`. So vocabulary indices start at 2 for real values.
  If cardinality > max_categorical_cardinality, keep only the top-N
  most frequent values; map the rest to __OOV__.
- Missing categoricals: map NaN / None to __MISSING__ index (1).
- Unseen categoricals (at transform time): map to __OOV__ index (0).
- Target handling:
  - Binary/multiclass: fit LabelEncoder on y_train.
  - Regression: fit StandardScaler on y_train.

```
def compute_metrics(
    y_true: np.ndarray,
    y_pred: np.ndarray,
    y_pred_proba: np.ndarray | None,
    task: str,
) -> dict[str, float]:
    """Compute all relevant metrics for a given task.

    Parameters
    ----------
    y_true : np.ndarray
        Ground truth labels/values.
    y_pred : np.ndarray
        Predicted labels (classification) or values (regression).
    y_pred_proba : np.ndarray or None
        Predicted probabilities. Shape (n,) for binary (probability
        of positive class) or (n, K) for multiclass. None for
        regression.
    task : str
        "binary", "multiclass", "regression"

    Returns
    -------
    dict[str, float]
        Keys depend on task:
```

```
from abc import ABC, abstractmethod

class EmbeddingModel(ABC):
    """Abstract base class for tabular embedding models.

    All embedding models must implement fit, encode, save, and load.
    The encode method extracts a fixed-length dense vector per sample
    that will be used as input to classical models.

    Subclasses must set self.embedding_dim after fit().
    """

    embedding_dim: int  # set after fit()

    @abstractmethod
    def fit(
        self,
        X_train: dict,
        y_train: np.ndarray,
        X_val: dict,
        y_val: np.ndarray,
        config: dict,
    ) -> "EmbeddingModel":
        """Train the embedding model.

        Parameters
        ----------
        X_train : dict
            Keys: "numeric" (np.ndarray, shape (n, n_num)),
                  "categorical" (np.ndarray, shape (n, n_cat), int64)
        y_train : np.ndarray
        X_val : dict (same format)
        y_val : np.ndarray
        config : dict
            Model hyperparameters.

        Returns
        -------
        self
        """

    @abstractmethod
    def encode(self, X: dict) -> np.ndarray:
        """Extract embeddings.

        Parameters
        ----------
        X : dict (same format as fit)

        Returns
        -------
        np.ndarray of shape (n_samples, self.embedding_dim)
        """

    @abstractmethod
    def save(self, path: str) -> None:
        """Save model state to disk."""

    @classmethod
    @abstractmethod
    def load(cls, path: str) -> "EmbeddingModel":
        """Load model state from disk."""

    def get_training_history(self) -> dict:
        """Return training history (losses, metrics per epoch).

        Default implementation returns empty dict. Subclasses should
        override.
        """
        return {}
```

```
class TabularMLP(nn.Module):
    """PyTorch module: MLP with categorical embedding tables.

    Architecture:
    1. Per-categorical-column embedding table (nn.Embedding)
    2. Numeric input: optional linear projection to match embedding dim
    3. Concatenate all embeddings + numeric projection
    4. MLP trunk: sequence of [Linear -> Norm? -> Activation -> Dropout]
    5. Output head: Linear -> output (for training loss)
    6. Embedding extraction point: output of last hidden layer
       (penultimate layer, before the output head)

    Parameters
    ----------
    n_numeric : int
    n_categories_per_column : list[int]
        Vocabulary size per categorical column.
    embedding_dim : int
        Dimension for each categorical embedding.
    hidden_dims : list[int]
        Width of each hidden layer.
    dropout : float
    activation : str
        "relu", "gelu", or "silu"
    use_layer_norm : bool
    task : str
        "binary", "multiclass", "regression"
    n_classes : int or None
        Number of classes (for classification output head).
    """

    def forward(self, x_numeric, x_categorical):
        """Full forward pass (for training). Returns task output."""

    def encode(self, x_numeric, x_categorical):
        """Forward pass up to penultimate layer. Returns embedding."""


class MLPEmbeddingModel(EmbeddingModel):
    """EmbeddingModel wrapper around TabularMLP.

    Handles training loop, early stopping, device management,
    and implements the full EmbeddingModel interface.

    Config keys (from Hyperopt search space):
        embedding_dim: int (32-256)
        hidden_dims: list[int] (derived from depth + width)
        dropout: float (0-0.5)
        activation: str
        use_layer_norm: bool
        lr: float
        weight_decay: float
        batch_size: int
        max_epochs: int
        patience: int
        device: str
    """

    def fit(self, X_train, y_train, X_val, y_val, config):
        """Train the MLP.

        Training loop:
        1. Build model from config
        2. Create DataLoaders
        3. Train with AdamW optimizer
        4. Early stopping on validation loss
        5. Restore best weights

        The validation metric used for early stopping is the task
        loss (not the downstream proxy metric -- that is handled
        by the HPO layer).
        """

    def encode(self, X):
        """Extract embeddings using the trained model.

        Sets model to eval mode, disables gradients, runs encode()
        in batches, concatenates, returns CPU numpy array.
        """

    def save(self, path): ...
    def load(cls, path): ...
```

**Key implementation details**:

- Loss function: BCEWithLogitsLoss (binary), CrossEntropyLoss
  (multiclass), MSELoss (regression).
- Early stopping: track validation loss. If no improvement for
  `patience` epochs, stop and restore best weights.
- Encoding: use `model.encode()` (penultimate layer output) in
  eval mode with `torch.no_grad()`.
- Batch encoding: process in batches to avoid OOM on large datasets.
  Default encoding batch size: 4096.
- Device: move model and data to config["device"].

```
class PytorchTabularEmbeddingModel(EmbeddingModel):
    """Adapter wrapping pytorch-tabular models as EmbeddingModel.

    This adapter translates between the hybrid_embed interface
    (preprocessed dict inputs) and pytorch-tabular's expected
    DataFrame + config interface. It handles:

    1. Converting preprocessed data back to DataFrame format
       (pytorch-tabular expects DataFrames)
    2. Building the pytorch-tabular model config from the HPO
       params dict
    3. Training via pytorch-tabular's Trainer
    4. Extracting embeddings from the model's backbone
       (penultimate layer output)

    Parameters
    ----------
    model_type : str
        Which pytorch-tabular model to use. Supported:
        - "tab_transformer" (TabTransformer, Huang et al. 2020)
        - "category_embedding" (CategoryEmbeddingModel)
        - "tabnet" (TabNet, Arik & Pfister 2019)
        - "gate" (GATE, Badirli et al. 2020)
        Additional models can be added by extending the
        _MODEL_CONFIG_MAP.

    schema : Schema
        Column type information.

    Config keys (common to all models):
        model_type: str
        lr: float
        weight_decay: float
        batch_size: int
        max_epochs: int
        patience: int
        device: str (mapped to pytorch-tabular trainer args)

    Config keys (TabTransformer-specific):
        input_embed_dim: int (32-128)
        num_heads: int (2-8)
        num_attn_blocks: int (2-6)
        attn_dropout: float (0-0.3)
        add_shared_embedding: bool
        shared_embedding_fraction: float (if shared embedding on)

    Config keys (other models):
        Model-specific params mapped to the corresponding
        pytorch-tabular model config class.
    """

    def __init__(self, model_type: str, schema):
        ...

    def fit(self, X_train, y_train, X_val, y_val, config):
        """Train via pytorch-tabular.

        Steps:
        1. Convert preprocessed dicts to DataFrames
        2. Build DataConfig, ModelConfig, TrainerConfig,
           OptimizerConfig from the config dict
        3. Instantiate TabularModel
        4. Call tabular_model.fit(train=train_df, validation=val_df)
        5. Store the trained model for later encoding
        """

    def encode(self, X):
        """Extract embeddings from the trained model's backbone.

        Uses pytorch-tabular's model to get the backbone output
        (before the final prediction head). This is the
        penultimate-layer representation.

        Returns numpy array of shape (n_samples, embedding_dim).
        """

    def save(self, path):
        """Save using pytorch-tabular's save_model method."""

    @classmethod
    def load(cls, path):
        """Load using pytorch-tabular's load_model method."""
```

**Key implementation details**:

- Data conversion: pytorch-tabular expects pandas DataFrames with
  column names. The adapter must reconstruct DataFrames from the
  preprocessed dict format. Store column names from schema.
- Target column: pytorch-tabular needs the target in the DataFrame.
  The adapter must add y as a column.
- Embedding extraction: pytorch-tabular models have a `backbone`
  attribute (or similar) that exposes the representation before the
  output head. Use a forward hook or the model's built-in method
  to extract these representations.
- Model config mapping: maintain a dict that maps model_type strings
  to their pytorch-tabular config classes (TabTransformerConfig,
  CategoryEmbeddingModelConfig, etc.).
- Early stopping: configure via pytorch-tabular's TrainerConfig
  (it uses PyTorch Lightning under the hood).
- Device handling: pytorch-tabular manages devices via its trainer.
  Pass the device preference through TrainerConfig.

```
def get_mlp_search_space() -> dict:
    """Return Hyperopt search space for MLP embedding model.

    Keys:
        embedding_dim: hp.quniform(32, 256, 16)  -> cast to int
        depth: hp.quniform(2, 8, 1) -> cast to int
        width: hp.quniform(128, 2048, 64) -> cast to int

    """Return Hyperopt search space for a pytorch-tabular model.

    Parameters

    Returns a Hyperopt space dict with model-specific keys.

    TabTransformer keys:

        num_heads: hp.choice([2, 4, 8])

        attn_dropout: hp.uniform(0, 0.3)

        shared_embedding_fraction: hp.uniform(0.25, 0.75)

    CategoryEmbedding keys:
        layers: derived from depth + width
        dropout: hp.uniform(0, 0.5)
        activation: hp.choice(["ReLU", "GELU", "SiLU"])
        lr, weight_decay, batch_size (same ranges)

    Additional model types follow similar patterns.
    """

def resolve_embedding_search_space(

    """

def sample_from_space(space: dict) -> dict:

def postprocess_mlp_params(params: dict) -> dict:
```

```
def hyperopt_search_embedding(
    X_train: dict,
    y_train: np.ndarray,
    X_val: dict,
    y_val: np.ndarray,
    task: str,
    n_numeric: int,
    n_categories_per_column: list[int],
    n_classes: int | None,
    embedding_step: EmbeddingStepConfig,
    proxy_model: str = "linear",
    max_trials: int = 50,
    max_epochs: int = 50,
    patience: int = 8,
    time_budget_seconds: float | None = None,
    master_seed: int = 42,
    device: str = "cpu",
    logger: LocalLogger | None = None,
) -> dict:
    """Run Hyperopt TPE search for the selected embedding model's HPs.

    Tunes the ONE model specified in embedding_step.model_type
    within its search space (user-provided or default).

    Per trial:
    1. Sample hyperparameters from the search space.
    2. Merge with embedding_step.fixed_params.
    3. Postprocess params.
    4. Build and train embedding model (with early stopping).
    5. Extract embeddings for train and val.
    6. Train proxy model on embeddings.
    7. Score proxy model on val embeddings.
    8. Log trial to logger (if provided).
    9. Return negative score for Hyperopt.

    Returns
    -------
    dict with keys:
        "best_params": dict (postprocessed best configuration)
        "best_score": float (best validation metric value)
        "n_trials_completed": int
        "total_time_seconds": float
    """
```

**Proxy model logic**:

```
def _build_proxy_model(task: str, proxy_type: str):
    """Build the proxy model used to evaluate embedding quality.

    proxy_type "linear":
        classification -> LogisticRegression(max_iter=1000)
        regression -> Ridge()

    proxy_type "lightgbm":
        classification -> LGBMClassifier(n_estimators=100, verbose=-1)
        regression -> LGBMRegressor(n_estimators=100, verbose=-1)
    """

def _evaluate_embeddings(
    E_train, y_train, E_val, y_val, task, proxy_type
) -> float:
    """Train proxy model on embeddings, return validation score.

    Uses the primary metric for the task (roc_auc, accuracy, rmse).
    For regression (where lower is better), negate so that higher
    is always better (Hyperopt minimizes, so we negate again in
    the objective).
    """
```

```
def run_experiment(
    X: pd.DataFrame,
    y: pd.Series | np.ndarray,
    task_config: TaskConfig,
    embedding_step: EmbeddingStepConfig,
    classical_step: ClassicalStepConfig,
    budget_config: BudgetConfig,
    run_config: RunConfig,
) -> RunResult:
    """Run full experiment pipeline.

    The user has already chosen ONE embedding model and ONE
    classical model. This function tunes their hyperparameters
    and evaluates via K-fold CV.

    Steps:
    1. seed_everything(run_config.master_seed)
    2. Validate data -> X_clean, validation_report
    3. Infer schema
    4. Generate CV splits
    5. HPO phase (on fold 0):
       a. Fit preprocessor on fold-0 train
       b. Transform fold-0 train/val
       c. Transform target
       d. Run embedding HPO for the selected model
       e. Using best embedding config:
          - Train embedding model on fold-0 train
          - Extract fold-0 train/val embeddings
          - Run classical HPO for the selected model
    6. Per-fold evaluation:
       For each fold:
       a. Fit fresh preprocessor on this fold's train
       b. Transform train, val, test
       c. Train embedding model with best config (early stop on val)
       d. Extract embeddings for train, val, test
       e. Train classical model with best config on train embeddings
          (with val for early stopping if applicable)
       f. Predict on test embeddings
       g. Compute metrics
       h. Save fold artifacts
    7. Aggregate metrics across folds
    8. Save summary and return RunResult

    Parameters
    ----------
    X : pd.DataFrame
    y : pd.Series or np.ndarray
    task_config : TaskConfig
    embedding_step : EmbeddingStepConfig
    classical_step : ClassicalStepConfig
    budget_config : BudgetConfig
    run_config : RunConfig

    Returns
    -------
    RunResult
    """
```
